import axios from "axios";
import { parseContent } from "@/src/firefit/RunParser";

/*

SELECT MIN(competition_id) as competition_id, MIN(duration) as duration, names
FROM results
WHERE names IN(
  ARRAY['Stefanie Reich']::varchar[],
  ARRAY['Till Schmidt']::varchar[],

)

export default defineEventHandler(async (event) => {
  const { data } = await axios.get("https://firefit-europe.eu/en/runs/");
  return parseContent(data);
});
